            \draw[thick, -stealth] (0,0) -- (2,0) node[midway, above] {\(b\)};
            \draw[thick, -stealth] (0,4) -- (0.8,4) node[right] {\(a\)};
        \end{tikzpicture}
    \end{center}

    \end{enumerate}
